Quick Start Example

# Validate an email address
curl -X POST "https://api.orbitcheck.io/v1/validate/email" \
  -H "Authorization: Bearer your_api_key" \
  -H "Content-Type: application/json" \
  -d '{"email": "user@example.com"}'

# Response
{
  "valid": true,
  "normalized": "user@example.com",
  "domain": "example.com",
  "mx_found": true,
  "disposable": false,
  "reason_codes": []
}

Order Risk Assessment

# Evaluate order for fraud risk
curl -X POST "https://api.orbitcheck.io/v1/orders/evaluate" \
  -H "Authorization: Bearer your_api_key" \
  -H "Content-Type: application/json" \
  -d '{
    "customer": {
      "email": "customer@example.com",
      "phone": "+1234567890",
      "name": "John Doe"
    },
    "order": {
      "amount": 99.99,
      "currency": "USD",
      "items": [...]
    }
  }'

# Response includes risk score, recommended action, and detailed analysis
